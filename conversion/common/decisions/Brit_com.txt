supply_communists = {
	is_shown = {
		exists = c:X10
		this = c:X10
		has_law = law_type:law_council_republic
		NOT = { has_variable = supplied_communists }
		exists = c:GBR
	}
	
	possible = {
		gold_reserves > 100000
	}
	
	when_taken = {
		add_treasury = -100000
		set_variable = {
			name = communist_uprising_planned
			days = 365
		}
		set_variable = {
			name = supplied_communists
		}
	}
	
	ai_chance = {
		base = 100
	}
}
