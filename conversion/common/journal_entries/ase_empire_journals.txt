je_house_of_cards = {
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	complete = {
		has_variable = ase_turmoil_france_over
		OR = {
			has_variable = ase_turmoil_america_over
			AND = {
				has_variable = ase_turmoil_north_america_over
				has_variable = ase_turmoil_south_america_over
			}
		}
		has_variable = ase_turmoil_china_over
		has_variable = ase_turmoil_caliph_over
		has_variable = ase_turmoil_monarchy_over
	}
	
	on_complete = {
		set_variable = house_of_cards_done
	}
	
	
	weight = 10000
	should_be_pinned_by_default = yes
}

je_ase_authority = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	complete = {
		has_variable = ase_ready
		OR = {
			NOT = { custom_tooltip = ase_min_authority }
			NOT = { custom_tooltip = ase_max_authority }
			var:ase_authority = 0
			var:ase_authority = 10
			has_variable = authority_hoc_done
		}
		NOT = { has_variable = authority_hoc_cooldown }
	}
	
	
	on_complete = {
		set_variable = ase_turmoil_monarchy_over
		if = {
			limit = {
				var:ase_authority = 0
			}
			trigger_event = ase_event.15
			change_variable = { 
				name = caliph_power
				subtract = 3
			}
			change_variable = { 
				name = colonial_power
				subtract = 3
			}
			change_variable = { 
				name = french_power
				subtract = 3
			}
			change_variable = { 
				name = chinese_power
				subtract = 3
			}
			activate_law = law_type:law_parliamentary_republic
			activate_law = law_type:law_census_voting
		}
		if = {
			limit = {
				var:ase_authority = 10
			}
			trigger_event = ase_event.13
			change_variable = { 
				name = caliph_power
				add = 3
			}
			change_variable = { 
				name = colonial_power
				add = 1
			}
			change_variable = { 
				name = french_power
				add = 1
			}
			change_variable = { 
				name = chinese_power
				add = 1
			}
		}
		if = {
			trigger_event = ase_event.14
			limit = {
				NOT = {
					var:ase_authority = 10
					var:ase_authority = 0
				}
			}
			change_variable = { 
				name = caliph_power
				subtract = 1
			}
			change_variable = { 
				name = colonial_power
				subtract = 1
			}
			change_variable = { 
				name = french_power
				subtract = 1
			}
			change_variable = { 
				name = chinese_power
				subtract = 1
			}
			activate_law = law_type:law_census_voting
		}
	}
	
	current_value = {
		value = var:ase_authority
	}
	
	goal_add_value = {
		add = 10
	}
	
	progressbar = yes
	
	weight = 10000
	should_be_pinned_by_default = yes
}

je_america_authority = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	complete = {
		has_variable = ase_ready
		OR = {
			NOT = { custom_tooltip = ase_min_america }
			NOT = { custom_tooltip = ase_max_america }
			var:colonial_power = 0
			var:colonial_power = 10
		}
	}
	
	on_complete = {
		if = {
			limit = {
				var:colonial_power = 0
			}
			change_variable = { 
				name = ase_authority
				subtract = 3
			}
			change_variable = { 
				name = caliph_power
				subtract = 3
			}
			set_variable = ase_turmoil_america_over
			c:GCO = {
				annex = c:NIC
				annex = c:X20
				annex = c:X28
				annex = c:X24
				change_tag = ARG
			}
			c:NEN = {
				annex = c:X29
				annex = c:WIN
				annex = c:X84
				annex = c:ONT
				change_tag = LOU
			}
			c:X42 = {
				annex = c:X83
				add_primary_culture = cu:neo-mexico_colony-faykinghian
			}
		}
		if = {
			limit = {
				var:colonial_power = 10
			}
			change_variable = { 
				name = ase_authority
				add = 2
			}
			change_variable = { 
				name = caliph_power
				add = 1
			}
			set_variable = ase_turmoil_america_over
		}
		if = {
			limit = {
				NOT = {
					var:colonial_power = 10
					var:colonial_power = 0
				}
			}
			change_variable = { 
				name = ase_authority
				subtract = 1
			}
			set_variable = ase_colonial_turmoil
			add_journal_entry = {
				type = je_north_favor
				target = this
			}
			add_journal_entry = {
				type = je_south_favor
				target = this
			}
			if = {
				limit = {
					var:colonial_power > 6
				}
				change_variable = { 
					name = north_favor
					add = 1
				}
				change_variable = { 
					name = south_favor
					add = 1
				}
			}
			c:X42 = {
				annex = c:X83
				add_primary_culture = cu:neo-mexico_colony-faykinghian
			}
			change_variable = {
				name = north_favor
				add = 2
			}
			change_variable = {
				name = south_favor
				add = 2
			}
		}
	}
	
	current_value = {
		value = var:colonial_power
	}
	
	goal_add_value = {
		add = 10
	}
	
	progressbar = yes
	weight = 10000
	should_be_pinned_by_default = yes
}
