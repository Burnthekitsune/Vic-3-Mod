je_france_authority = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	complete = {
		has_variable = ase_ready
		has_variable = ase_ready
		OR = {
			NOT = { custom_tooltip = ase_min_france }
			NOT = { custom_tooltip = ase_max_france }
			var:french_power = 0
			var:french_power = 10
		}
	}
	
	on_complete = {
		set_variable = ase_turmoil_france_over
		if = {
			limit = {
				has_variable = france_triumph
			}
			annex = c:FRA
			annex = c:BRI
		}
		if = {
			limit = {
				has_variable = france_failure
			}
			c:FRA = {
				make_independent = true
			}
			c:BRI = {
				make_independent = true
			}
			# release Shamali
		}
		if = {
			limit = {
				has_variable = france_compromise
			}
			FRA = {
				make_independent = true
			}
		}
	}
	
	current_value = {
		value = var:french_power
	}
	
	goal_add_value = {
		add = 10
	}
	
	progressbar = yes
	weight = 10000
	should_be_pinned_by_default = yes
}
