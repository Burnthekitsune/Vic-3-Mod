je_yucatan_unification = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	is_shown_when_inactive = {
		country_has_primary_culture = cu:mayan
		OR = {
			country_tier = kingdom
			country_tier = principality
			country_tier = grand_principality
			country_tier = city_state
		}
	}

	possible = {
		country_rank >= rank_value:minor_power
		has_technology_researched = nationalism
		OR = {
			is_subject = yes
			is_subject = no
		}
	}

	complete = {
		hidden_trigger = {
			has_variable = je_unification_progress
		}
		calc_true_if = {
			owns_entire_state_region = STATE_YUCATAN
			owns_entire_state_region = STATE_GUATEMALA
			owns_entire_state_region = STATE_CHIAPAS
			owns_entire_state_region = STATE_VERACRUZ
			owns_entire_state_region = STATE_OAXACA
			amount >= 4
		}
	}

	on_monthly_pulse = {
		events = {
			yucatan_unification.2 # Biscarp
		}
		effect = {
			if = { 
				limit = { NOT = { has_variable = je_unification_progress } } 
				set_variable = je_unification_progress
			}
		}
	}

	on_complete = {
		trigger_event = { id = yucatan_unification.1 } # Yucatan Confederation
		remove_variable = je_unification_progress
	}


	weight = 100
}

je_mexican_unification = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	is_shown_when_inactive = {
		OR = {
			country_has_primary_culture = cu:maya
			country_has_primary_culture = cu:zapotec
			country_has_primary_culture = cu:neo-mexico_colony-north_american_alshamali
			country_has_primary_culture = cu:neo-texas_colony-north_american_alshamali
			country_has_primary_culture = cu:apache
			country_has_primary_culture = cu:navajo
		}
	}

	possible = {
		country_rank >= rank_value:major_power
		has_technology_researched = romanticism
		OR = {
			is_subject = yes
			is_subject = no
		}
	}

	complete = {
		hidden_trigger = {
			has_variable = je_unification_progress
		}
		calc_true_if = {
			amount >= 16
			owns_entire_state_region = STATE_YUCATAN
			owns_entire_state_region = STATE_GUATEMALA
			owns_entire_state_region = STATE_CHIAPAS
			owns_entire_state_region = STATE_VERACRUZ
			owns_entire_state_region = STATE_OAXACA
			owns_entire_state_region = STATE_GUERRERO
			owns_entire_state_region = STATE_MEXICO
			owns_entire_state_region = STATE_JALISCO
			owns_entire_state_region = STATE_BAJIO
			owns_entire_state_region = STATE_RIO_GRANDE
			owns_entire_state_region = STATE_ZACATECAS
			owns_entire_state_region = STATE_DURANGO
			owns_entire_state_region = STATE_CHIHUAHUA
			owns_entire_state_region = STATE_SINALOA
			owns_entire_state_region = STATE_SONORA
			owns_entire_state_region = STATE_BAJA_CALIFORNIA
			owns_entire_state_region = STATE_ARIZONA
			owns_entire_state_region = STATE_NEW_MEXICO
			owns_entire_state_region = STATE_TEXAS
		}
	}

	on_complete = {
		trigger_event = { id = thunderdome.1 } # Mexican Unification
		remove_variable = je_unification_progress
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_mexican_status
				trigger = {
					OR = {
						country_tier = kingdom
						country_tier = principality
					}
				}
			}
		}
	}

	invalid = {
		country_has_primary_culture = cu:hungarian
	}

	weight = 100
}
